# 1.二分查找

二分查找用于有序数组，若无序要先排序

示例：

**1.判断数组中是否含有某元素**

```C++
#include<bits/stdc++.h>
using namespace std;
int arr[5];
int main() {
	int f = -1;
	int key, left = 0,right=4;
	for (int i = 0; i < 5; i++)
	{
		cin >> arr[i];
	}
	cin >> key;
	sort(arr, arr + 5);
	while (left <= right)
	{
		int mid = (left + right) / 2;
		if (arr[mid] < key) left = mid + 1;
		if (arr[mid] > key) right = mid - 1;
		if (arr[mid] == key) 
		{
			f = mid;
			break;
		}
	}
	if (f == -1) cout << "not found!" << endl;
	else cout << f << endl;//输出该元素在数组中排序后的位置
	return 0;
}
```

不足之处：当该元素在数组不止一个时，输出的位置可能是该元素的任意一个位置，**并非首位。**

**2.判断数组中该元素出现的次数**

```C++
#include<bits/stdc++.h>
using namespace std;
const int n = 1e5 + 1;
int arr[n], key,m;
int firstKey(int arr[], int key)
{
	int left = 0, right = m - 1, result = -1;
	while (left <= right)
	{
		int mid = (left + right) / 2;
		if (arr[mid] == key)
		{
			result = mid;
			right = mid - 1;
		}
		if (arr[mid] > key)
		{
			right = mid - 1;
		}
		if (arr[mid] < key)
		{
			left = mid + 1;
		}
	}
	return result;
}
int lastKey(int arr[], int key)
{
	int left = 0, right = m - 1, result = -1;
	while (left <= right)
	{
		int mid = (left + right) / 2;
		if (arr[mid] == key)
		{
			result = mid;
			left = mid + 1;
		}
		if (arr[mid] > key)
		{
			right = mid - 1;
		}
		if (arr[mid] < key)
		{
			left = mid + 1;
		}
	}
	return result;
}
int main()
{
	cin >> m;
	for (int i = 0; i < m; i++)
	{
		cin >> arr[i];
	}
	cin >> key;
	sort(arr, arr + m);
	int first = firstKey(arr, key);
	int last = lastKey(arr, key);
	int count = last - first + 1;
	if (count == 0)  cout << "Not found!" << endl;
	else cout << count << endl;
	return 0;
}
```
